## ğŸ“Š Data Model for Reports

### **Database Schema for Reporting**

```sql
-- Admin reporting tables
CREATE TABLE admin_reports (
  id UUID PRIMARY KEY,
  school_id UUID NOT NULL,
  report_type VARCHAR(50) NOT NULL, -- 'executive', 'compliance', 'roi', 'teacher_performance'
  report_period VARCHAR(20) NOT NULL, -- 'daily', 'weekly', 'monthly', 'quarterly'
  generated_by UUID NOT NULL,
  generated_at TIMESTAMP DEFAULT NOW(),
  report_data JSONB NOT NULL,
  status VARCHAR(20) DEFAULT 'generated',
  file_path VARCHAR(255),
  expires_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Report metrics cache
CREATE TABLE report_metrics (
  id UUID PRIMARY KEY,
  school_id UUID NOT NULL,
  metric_type VARCHAR(50) NOT NULL,
  metric_name VARCHAR(100) NOT NULL,
  metric_value DECIMAL(10,2),
  metric_data JSONB,
  period_start DATE NOT NULL,
  period_end DATE NOT NULL,
  calculated_at TIMESTAMP DEFAULT NOW(),
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(school_id, metric_type, metric_name, period_start, period_end)
);

-- Admin actions log
CREATE TABLE admin_actions (
  id UUID PRIMARY KEY,
  admin_id UUID NOT NULL,
  school_id UUID NOT NULL,
  action VARCHAR(50) NOT NULL,
  target_type VARCHAR(20) NOT NULL, -- 'TEACHER', 'SYSTEM', 'MESSAGE', 'PARENTS'
  target_id UUID,
  reason TEXT,
  metadata JSONB,
  ip_address INET,
  user_agent TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- System pause tracking
CREATE TABLE system_pauses (
  id UUID PRIMARY KEY,
  school_id UUID NOT NULL,
  admin_id UUID NOT NULL,
  reason TEXT NOT NULL,
  paused_at TIMESTAMP NOT NULL,
  resume_at TIMESTAMP,
  resumed_at TIMESTAMP,
  resumed_by UUID,
  resume_reason TEXT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Report subscriptions
CREATE TABLE report_subscriptions (
  id UUID PRIMARY KEY,
  school_id UUID NOT NULL,
  admin_id UUID NOT NULL,
  report_type VARCHAR(50) NOT NULL,
  frequency VARCHAR(20) NOT NULL, -- 'daily', 'weekly', 'monthly'
  delivery_method VARCHAR(20) DEFAULT 'email', -- 'email', 'dashboard', 'both'
  email_address VARCHAR(255),
  is_active BOOLEAN DEFAULT true,
  last_sent TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);
```
